baidu-star算法题-地铁

baidu-star算法题-地铁

02 Jan 2013 By amyangfei

来源于百度之星算法比赛2012年12月23号竞赛题目三，一道有向图节点出度、入度相关的题目

题目来源：百度之星-2012年12月23号竞赛题目三-地铁

思路：地图是一个有向图，需要从M条边中删除M-K条有向边。下面将一个点的出度与入度差的绝对值记为该点的“度”。由于只删边不增加边，那么如果希望降低系统的最大度，那么具有最大度的顶点一定需要删除与其相关的边。考虑使用贪心算法，每次选择删除边的规则如下：

  1. 选出具有最大度的所有顶点，定义一个待删除相关节点最大度lastdegmax。对应每一个顶点u，如果u的出度大于入度，那么对应每一条(u,v),若删除(u,v)后会降低v的度，并且v的度大于lastdegmax，那么将(u,v)设置为将要删除的缓冲区(待删除缓冲区长度为1)。如果u的入度大于出度，操作类似。
  2. 如果1中删除任何一条(u,v)或(v,u)都会增加v的度，那么选择选择具有最小度的v，将(u,v)或(v,u)加入待删除缓冲区。
  3. 缓冲区中的边即为该轮迭代中将要删除的边。

实现代码如下：

#include <iostream>
#include <vector>
#include <map>
#include <cmath>
using namespace::std;

bool SortByDeg(const pair<int,int> &v1, const pair<int,int> &v2) {
    return abs(v1.second) > abs(v2.second); //desc sort
}

void SortWrap(vector<pair<int,int> > &deg, map<int, int> &degmap) {
    sort(deg.begin(), deg.end(), SortByDeg);
    for (int i = 0; i < deg.size(); i++) {
        degmap[deg[i].first] = i;
    }
}

void deldeg(vector<int> &deldeg, int u) {
    for (vector<int>::iterator uiter = deldeg.begin(); uiter != deldeg.end(); uiter++) {
        if (*uiter == u) {
            deldeg.erase(uiter);
            break;
        }
    }
}

void update(vector<int> &outdeg, vector<int> &indeg, vector<pair<int,int> > &deg, map<int, int> &degmap, int u, int v) {
    deldeg(outdeg, v);
    deldeg(indeg, u);
    int ret = 0;
    for (vector<pair<int, int> >::iterator iter = deg.begin(); iter != deg.end(); iter++) {
        if (ret == 2)
            break;
        if ((*iter).first == u) {
            (*iter).second -= 1;
            ret += 1;
        }
        else if ((*iter).first == v) {
            (*iter).second += 1;
            ret += 1;
        }
    }
    SortWrap(deg, degmap);
}

void setdelbuf(pair<int,int> &delbuf, int u, int v, int handleOut) {
    delbuf.first = handleOut>0 ? u : v;
    delbuf.second = handleOut>0 ? v : u;
}

void oneSelect(vector<int> &outdeg, vector<int> &indeg, vector<pair<int,int> > &deg, map<int, int> &degmap, pair<int,int> &delbuf, int u, int *lastdegmax, int *lastdegmin, bool *minusmax, int handleOut) {
    vector<int> handleDeg;
    if (handleOut == 1)
        handleDeg = outdeg;
    else
        handleDeg = indeg;
    for (vector<int>::iterator viter = handleDeg.begin(); viter != handleDeg.end(); viter++) {
        if(deg.at(degmap[*viter]).second * handleOut < 0) {
            if (abs(deg.at(degmap[*viter]).second) > *lastdegmax) {
                *lastdegmax = abs(deg.at(degmap[*viter]).second);
                setdelbuf(delbuf, u, *viter, handleOut);
                *minusmax = true;
            }
        }
        else if (!(*minusmax) ){
            if( abs(deg.at(degmap[*viter]).second) < *lastdegmin ) {
                setdelbuf(delbuf, u, *viter, handleOut);
                *lastdegmin = abs(deg.at(degmap[*viter]).second);
            }
        }
    }
}

void cutSelect(vector<int> *outdeg, vector<int> *indeg, vector<pair<int,int> > &deg, map<int, int> &degmap, pair<int,int> &delbuf) {
    int degmax = abs(deg.at(0).second);
    int lastdegmax = -1;
    int lastdegmin = degmax;
    bool minusmax = false;
    for (int j = 0; j<deg.size() && abs(deg.at(j).second)==degmax; j++) {
        int handleOut = deg.at(j).second>0 ? 1 : -1;
        oneSelect(outdeg[deg.at(j).first], indeg[deg.at(j).first], deg, degmap, delbuf, deg.at(j).first, &lastdegmax, &lastdegmin, &minusmax, handleOut);
    }
}

int solve(int n, int m, int k, vector<pair<int,int> > &way) {
    vector<int> *outdeg = new vector<int>[n];
    vector<int> *indeg = new vector<int>[n];
    for (vector<pair<int, int> >::iterator iter = way.begin(); iter != way.end(); iter++) {
        outdeg[(*iter).first].push_back((*iter).second);
        indeg[(*iter).second].push_back((*iter).first);
    }
    vector<pair<int,int> > deg;
    map<int, int> degmap;
    for (int i = 0; i < n; i++) {
        deg.push_back(make_pair(i, outdeg[i].size()-indeg[i].size()));
    }
    SortWrap(deg, degmap);
    
    for (int i = 0; i < m-k; i++) {
        pair<int, int> delbuf = make_pair(-1, -1);
        cutSelect(outdeg, indeg, deg, degmap, delbuf);
        update(outdeg[delbuf.first], indeg[delbuf.second], deg, degmap, delbuf.first, delbuf.second);
    }
    
    return abs(deg.at(0).second);
}

int main(int argc, const char * argv[]) {
    int T;
    cin >> T;
    vector<pair<int, int> > *way =new vector<pair<int, int> >[T];
    int *n = new int[T];
    int *m = new int[T];
    int *k = new int[T];
    for (int i = 0; i < T; i++) {
        cin >> n[i] >> m[i] >> k[i];
        for (int j = 0; j < m[i]; j++) {
            int a, b;
            cin >> a >> b;
            way[i].push_back(make_pair(a-1, b-1));
        }
    }
    for (int i = 0; i < T; i++) {
        cout << "Case " << (i+1) << ": " << solve(n[i], m[i], k[i], way[i]) << endl;
    }
    return 0;
}
